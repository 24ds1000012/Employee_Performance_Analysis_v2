import os
import base64
from io import StringIO, BytesIO
import pandas as pd
import matplotlib.pyplot as plt
from textwrap import dedent

# ---- 1) Load the employee data ----
sample_csv = """employee_id,department,region,performance_score,years_experience,satisfaction_rating
EMP001,HR,Asia Pacific,92.86,8,4.3
EMP002,IT,Europe,60.71,15,3.6
EMP003,Operations,Europe,73.56,7,4
EMP004,Operations,Latin America,87.61,14,4.8
EMP005,IT,North America,76.43,9,3.1
"""

csv_path = "/mnt/data/employee_performance.csv"
if os.path.exists(csv_path):
    df = pd.read_csv(csv_path)
    data_source_note = f"Loaded dataset from {csv_path}."
else:
    df = pd.read_csv(StringIO(sample_csv))
    data_source_note = "Loaded sample dataset embedded in the script (5 rows)."

print(data_source_note)

# ---- 2) Frequency count for the "Finance" department ----
finance_count = (df["department"] == "Finance").sum()
print(f'Frequency count for "Finance" department: {finance_count}')

# ---- 3) Create the department distribution plot ----
dept_counts = df["department"].value_counts()

plt.figure(figsize=(8, 5))
plt.bar(dept_counts.index, dept_counts.values)
plt.title("Distribution of Employees by Department")
plt.xlabel("Department")
plt.ylabel("Count")
plt.xticks(rotation=45, ha="right")
plt.tight_layout()

plot_png_path = "/mnt/data/department_distribution.png"
plt.savefig(plot_png_path, bbox_inches="tight")
plt.close()
